ButtonsPressed()
{
     ; Solange beide Maustasten gedr端ckt werden, halte MButton gedr端ckt
    while (GetKeyState("LButton", "P") && GetKeyState("RButton", "P"))
    {
        Click("Middle", "D")  ; Mittlere Maustaste gedr端ckt halten
        Sleep(50)  ; Eine kurze Pause, um die CPU-Last zu reduzieren
    }
    Click("Middle", "U")  ; Mittlere Maustaste loslassen, wenn eine der Maustasten nicht mehr gedr端ckt wird
    ; Aufbrechen der Schleife, damit die linke Maustaste normal funktioniert
    while GetKeyState("LButton", "P") || GetKeyState("RButton", "P")
    {
        Sleep(10)  ; Warten, bis beide Tasten losgelassen werden
    }
}

~LButton & RButton:: 
    {
    ButtonsPressed()
    }

~RButton & LButton:: 
    {
    ButtonsPressed()
    }
    
return
